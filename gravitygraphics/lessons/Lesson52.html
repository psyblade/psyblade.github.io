Lesson 52: Functions With Parameters

You can also pass parameters into your functions. Let's say you want to give the computer a length and ask it to calculate the percentage of the length based on a random percentage.

gr();
function printRandomPercentage(length) {
	var number = random(1,100);
	var percentage_as_a_decimal = number/100;
	text(number + "% = " + number + "/100 = " + (number / 100));
	text(number + "% of the length " + length + " = " + percentage_as_a_decimal * length);
}
printRandomPercentage(length);
